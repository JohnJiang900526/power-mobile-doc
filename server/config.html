# 配置config.js

打开PowerMobile2文件夹 修改PowerMobile2\static\config\config.js文件。

# SITE_SESTION

SITE_SESTION是你当前项目的地址【必须配置】，项目地址的最后面不要带"/"， 注意站点地址要加请求协议。

  // 对的配置
  var SITE_SESTION = "http://xxx.com:7109";

  // 错的配置1 最后面不要带"/"
  var SITE_SESTION = "http://xxx.com:7109/";

  // 错的配置2 地址要加请求协议
  var SITE_SESTION = "xxx.com:7109/";

# NODE_SITE

NODE_SITE 代表的是nodeServer的地址【必须配置】，node服务器的站点，必须配置， 如果node服务没有部署，请参考node 服务端配置信息。此字段要在链接的后面添加"/"， 注意nodeServer站点地址要加请求协议

  // 对的配置
  var NODE_SITE = "http://xxx.com:3001/";

  // 错的配置1 此字段要在链接的后面添加"/"
  var NODE_SITE = "http://xxx.com:3001";

  // 错的配置2 注意nodeServer站点地址要加请求协议
  var NODE_SITE = "xxx.com:3001";

# DEBUG

DEBUG 字段用来辅助在移动端接口报错的时候，开启调试模式，

如果开启DEBUG，会显示vconsole, 通过vconsole可以查看到

对应接口是否正常运行。

  var DEBUG = false; // 默认false 可以设置为true

# OuterWebSite

// 外部站点连接
// 当此链接存在的时候，审批流程提交之后会触发
// 整个页面会跳转到设置的这个页面中
// 比如：https://www.baidu.com
var OuterWebSite = "";

# tabsData

此变量代表要激活的主tab页，默认配置信息如下：

var tabsData = [
  {
    to: "/message",
    name: "消息",
    icon: "icon-message"
  },
  {
    to: "/dynamics",
    name: "动态",
    icon: "icon-compass"
  },
  {
    to: "/portal",
    name: "门户",
    icon: "icon-project2"
  },
  {
    to: "/work",
    name: "工作台",
    icon: "icon-work"
  },
  {
    to: "/personal",
    name: "我",
    icon: "icon-mine"
  }
];

上面的默认字段和下面默认的tab页签一一对应。

如果想禁止其中一个tab页签的话，可以在上面的默认配置中去掉。比如：禁止掉动态和门户

如果数组 tabsData为空null或者[],上图中的按钮全部显示，

如果不想要其中的一个tab页签，可以在默认的tabsData中去掉对应的项，

但是必须保证tabsData不为null或者[]。

var tabsData = [
  {
    to: "/message",
    name: "消息",
    icon: "icon-message"
  },
  // {
  //   to: "/dynamics",
  //   name: "动态",
  //   icon: "icon-compass"
  // },
  // {
  //   to: "/portal",
  //   name: "门户",
  //   icon: "icon-project2"
  // },
  {
    to: "/work",
    name: "工作台",
    icon: "icon-work"
  },
  {
    to: "/personal",
    name: "我",
    icon: "icon-mine"
  }
];

# defaultActiveTab

此变量默认是字符串 "/message"，如果想改变此默认配置项，需要改变此变量的值， 此变量的值的取值区间是tabsData数据中的to字段的所有值。

// 配置默认显示的tab页
var defaultActiveTab = "/message";

# useChatModule

此变量的意思是是否启用聊天功能。此变量默认是false,表示不启用，如果设置为true， 将会启用聊天功能。聊天功能只能在APP环境使用，在微信环境中，不能使用。

// 是否启用聊天
var useChatModule = false; // true | false

# showUrlWhenNoConfig

此变量的意思是表单没有配置文件的时候，只显示详情页签， 默认是false 代表显示详情和所有子表，也包括附件， 如果设置成true, 只显示详情tab页签。

// 表单没有配置文件的时候，只显示详情页签
// 默认是false 代表显示详情和所有子表，也包括附件
// 如果设置成true, 只显示详情tab页签
var showUrlWhenNoConfig = false;

# UploadHumanAvatar

此变量的意思是是否启用个人中心上传人员头像的功能，启用true, 不启用 false。

// 个人中心 上传人员头像 启用true, 不启用 false
var UploadHumanAvatar = false;

# loginLogoUrl

此变量的意思是登录页的logo地址，默认"/PowerMobile2/static/img/login-logo.png"， 可以根据自己想要的logo进行配置。如果没有配置，logo将不会显示。

// 登录logo
var loginLogoUrl = "/PowerMobile2/static/img/login-logo.png";

# loginSubText

此变量的意思是设置登录页logo下方的问题，可以根据自己想要的名称进行修改，默认值为"项目管理平台移动应用"。

var loginSubText = "项目管理平台移动应用";

# defaultProjectAvatar

此变量的意思是配置默认项目头像，项目没有项目头像的情况下显示, 默认是 "/PowerMobile2/static/img/user1.png"

var defaultProjectAvatar = "/PowerMobile2/static/img/user1.png";

# allowMultipleSite

此变量的意思是是否允许审批消息进行跨站点聚合，默认值是false，一般不推荐使用跨站点聚合，使用切换站点就可以了。如果想开启，设置true。

// 是否允许跨站点聚合
var allowMultipleSite = false;

# hideWorkFlowAttach

是否隐藏审批流上传附件的功能

// 是否禁止流程上传附件功能
var hideWorkFlowAttach = false;

# useOuterWeb

是否使用外部表单，也就是表单中只嵌入PC转的html pho文件或者htm文件

// 使用外部链接：针对表单
var useOuterWeb = false;

# shortReturnEnable

驳回的简单模式 驳回时，前面已有多人审批，默认第一个节点,隐藏”直接回到我手里”,默认就是”按原步骤审批” 如果启用的话设置 true 否则设置false

// 驳回时，前面已有多人审批，默认第一个节点
// 隐藏”直接回到我手里”,默认就是”按原步骤审批”
var shortReturnEnable = false;

# useNewFormFlow

使用新版分支审批流程

var useNewFormFlow = false; 
// 流程做了升级
// 如果想用新的流程
// 设置true 默认是false

# appUploadMethod

// APP上传模式 
// 新增日期 2020-11-30
// Native 表示使用原生APP程序上传
// Dcloud 表示使用Dcloud第三方集成程序上传
// 如果app经常上传失败，特别是安卓app 建议使用 Dcloud 模式
// Dcloud 模式 需要最新的后端补丁
var appUploadMethod = "Native"; // Dcloud、Native

# customizeDynamicsUrl

// 自定义动态页面 默认为空的字符串 ""
// 新增日期 2020-11-30
// 仅仅在前端的生产环境中使用 在开发环境中不能使用
// 1.可以设置静态页面 "/PowerMobile2/test.html";
// 2.可以设置窗体页面 "/Mobile/ValidForm/窗体的FormId/edit/"
//   窗体例子 "/Mobile/ValidForm/8e3b012a-299a-9aca-c06e-b1954a9f043b/edit/";
// 3.可以设置表单页面 "/Mobile/ValidForm/表单的FormId/edit/表单的KeyValue/";
//   表单例子 "/Mobile/ValidForm/90944435-3671-4ab0-9174-5f8546638615/edit/44735c0c-c105-40a9-a23b-7638b175293f/";
var customizeDynamicsUrl = "";

# WindowReadOnly

// 全局模式下 
// WindowReadOnly 为true 代表
// 所有的窗体都是只读模式
// 如果对应的窗体没有配置WindowReadOnly 来修改 全局的WindowReadOnly
// 起效的是全局配置 如果窗体有对应配置WindowReadOnly 起效的是窗体的配置
var WindowReadOnly = false; // 默认false  [false, true]

# NotSaveForm

// 全局模式下 
// NotSaveForm 为true 代表
// 所有的表单都是无保存按钮方式
// 如果对应的表单没有配置NotSaveForm 来修改 全局的NotSaveForm
// 起效的是全局配置 如果表单有对应配置NotSaveForm 起效的是表单的配置
var NotSaveForm = false;
