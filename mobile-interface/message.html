# 标记通知消息已读接口

$.ajax({
  url: "/Mobile/notifyStatus",
  type: "post",
  dataType: "json",
  data: {
    token: "", // 登录接口中返回的token
    notifyId: "", //  
    // "true" | "false" 当设置为"true"的时候，标记所有通知已读。此字段是非必须字段，
    // 当设置为"true"时，忽略notifyId参数的值，readall:"true"的优先级大于notifyId字段，
    // 两个字段不是and关系
    // 标记所有已读
    readall: "", 
  },
  success: function () {
    
  },
  error: function () {

  }
});
