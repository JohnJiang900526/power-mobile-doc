# 窗体的详细配置

{
  "formId" : "", // 表单的formId(需要打开另外的表单的话)
  "keyword" : "", // 关键词
  "keywordType" : "", // BO(关键词类型BO | ViewEntity)
  "keyField" : "", // 主键, 默认Id
  "sequField": "Sequ desc", // 排序字段及方式(Sequ desc)
  "regHumField" : "", // 录入人名称字段
  "regHumIdField" : "", // 录入人Id字段
  "regDateField" : "", // 录入时间字段
  "hideAvatar": true, // 是否隐藏头像 默认false 
  "RegDateType": "yyyy-MM-dd", // 显示起草日期的格式 默认yyyy-MM-dd HH:mm:ss
  "WindowReadOnly": false, // 启用只读模式 会消除窗体下面的新增和删除按钮 [true, false] 默认false
  // filter 指的是过滤条件， filter可以是数组，也可以是对象
  // filter中 opt为=的时候 会在新增表单的时候自动添加到表单里面
  // filter 会在窗体中执行过滤 会在新增表单的时候自动新增属性(默认赋值)
  // 下面是filter数组的演示
  "filter" :[{
    "name":"", // 默认过滤字段名
    // opt条件语句
    // 如果是等于则"=", value直接值对应的值即可
    
    // 如果是模糊查询则"like", 
    // value中的值要符合sql的规范，比如("%123%", "%123", "123%")，一般推荐使用"%123%"这种模式
    
    // 如果是<, >, <=, >= 号，注意value的值要和name字段中的数据库字段类型保持一致

    // 如果是in value值为 '1,2'
    "opt":"=", 
    "value":"" // 默认过滤值
  }],
  // 下面是filter对象的演示
  "filter" :{
    "name":"", // 默认过滤字段名
    "opt":"=",
    "value":"" // 默认过滤值
  },

  // windowFilter 属性是窗体专用过滤属性，如果是在窗体是执行过滤 强烈建议在此属性中添加过滤条件
  // windowFilter 指的是过滤条件， windowFilter可以是数组，也可以是对象
  // 下面是windowFilter数组的演示 属性和filter一样，不在赘述
  "windowFilter": [
    {
      "name": "expensetype",
      "opt": "=",
      "value": "其他费用" // 默认过滤值 支持PC session默认参数
    }
  ],
  // 下面是windowFilter对象的演示
  "windowFilter" :{
    "name":"", // 默认过滤字段名
    "opt":"=",
    "value":"" // 默认过滤值 支持PC session默认参数
  },


  // 事件定义部分 不需要的话可以不用配置
  // 注意 beforeLoad， afterLoad， 还有callback全部都是字符串
  "jsContent": {
     "beforeLoad": "", // string 加载前事件 "function () {}" 
     "afterLoad": "", // string 加载后事件 "function () {}",
     // 自定义事件
     "customizeEvents": [
		{
			"name": "窗体自定义事件1",
      		"callback": "function () {}" // string
		},
		{
			"name": "窗体自定义事件2",
      		"callback": "function () {}" // string
		}
	]
  },
  "showFields": [
	  {
	  "name":"", // 要显示的字段名
	  "label":"", // 字段描述
	  //controlType代表的是 类型 text(默认普通文本) | number(数字) | 
	  //checkbox(选择框) |  datePicker(日期选择器) | select(下拉框) 
	  "controlType": "", 
	  // format 格式化规则 ''(默认) | n2 (数字千分符) | rate(数字 税率 小数点8位) | 
	  // yyyy-MM-dd (日期格式化) | boolean|number|string (针对checkbox)
	  "format":"",
	  // showPlace 字段显示的位置，search,tag，title，left，right，center
	  "showPlace":""
	  }
  ]
}

上面json中，controlType和format配合使用的，先controlType后format

  1. controlType 之text类型
{
	"controlType": "text",
	"format":"",// 如果controlType是text类型， format字段直接设置空的字符串即可
}
  1. controlType 之number类型
{
	"controlType": "number",
	"format":"",// 如果controlType是number类型， format字段可以设置为"n2" 或者 ""
}
  1. controlType 之checkbox类型
{
	"controlType": "checkbox",
	"format":"",// 如果controlType是checkbox类型， format字段可以设置为"boolean" | "string" | "number"
}
  1. controlType 之datePicker类型
{
	"controlType": "datePicker",
	"format":"",
	// 如果controlType是datePicker类型， 
	// format字段可以设置为
	// "yyyy-MM-dd HH:mm:ss" 
	// "yyyy-MM-ddTHH:mm:ss" 
	// "yyyy-MM-dd" 
}
  1. controlType 之select类型
{
	"controlType": "select",
	"format":"",
	// 如果controlType是select类型， 
	// format字段可以设置为""
	// 但是必须保证在PC端窗体对应的表单中的comboboxdata中要有对应的下拉框的配置
}
