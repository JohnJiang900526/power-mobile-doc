# 单点登录 页面说明

var single_url = "你的站点";
var url = "登录成功之后跳转的url";

single_url +="/PowerMobile2/static/singleLogin.html" 

single_url += "?UserCode=" + "单点登录人";

single_url += "&url=" + encodeURIComponent(url);

// 外站点执行跳转
window.location.href = single_url;

// 具体的逻辑见静态页面
