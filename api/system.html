# 系统方法 API

系统的方法很多，下面只展示几种常用的方法。

# 1. 调用二维码

// 1.1在vue的程序中调用
new window.PowerApp.system.ScanQRCode((result) => {
  // result 是二维码扫描的结果
  this.address = result;
});


// 1.2在vue的程序中iframe嵌入页面的调用
var that = this;
window.parent.PowerApp && new window.parent.PowerApp.system.ScanQRCode(function (result) {
  // result 是二维码扫描的结果
  that.address = result;
});

# 2. 上传附件

// 2.1在vue的程序中调用
// 参数：token, KeyValue, KeyWord, successCallBack, errorCallBack
new window.PowerApp.system.UploadFile(
  token, 
  KeyValue, 
  KeyWord, 
  (res) => {
  // successCallBack

  }, (err) => {
  // errorCallBack

});


// 2.2在vue的程序中iframe嵌入页面的调用
var that = this;
window.parent.PowerApp && new window.PowerApp.system.UploadFile(
  token, 
  KeyValue, 
  KeyWord, 
  function (res) {
    // successCallBack
  }, function (err) {
    // errorCallBack
  }
);

# 3. alert

// 3.1在vue的程序中调用
// title, message, callback(选填)
window.PowerApp.system.alert("title", "提示信息", () => {
  // 回调函数
});


// 3.2在vue的程序中iframe嵌入页面的调用
window.parent.PowerApp && window.PowerApp.system.alert("title", "提示信息", function () {
  // 回调函数
});

# 4. confirm

// 4.1在vue的程序中调用
// ttitle, message, cbSucc, cbCancel
window.PowerApp.system.confirm("title", "提示信息", () => {
  // cbSucc回调函数
}, () => {
  // cbCancel回调函数
});


// 4.2在vue的程序中iframe嵌入页面的调用
window.parent.PowerApp && window.PowerApp.system.confirm("title", "提示信息", function () {
  // cbSucc回调函数
}, function () {
  // cbCancel回调函数
});

# 5. loading

// 5.1在vue的程序中调用
// ttitle(选填)
window.PowerApp.system.loading("加载中...");


// 5.2在vue的程序中iframe嵌入页面的调用
window.parent.PowerApp && window.PowerApp.system.loading("加载中...");

# 6. closeLoading

// 5.1在vue的程序中调用
// ttitle
window.PowerApp.system.closeLoading();


// 5.2在vue的程序中iframe嵌入页面的调用
window.parent.PowerApp && window.PowerApp.system.closeLoading();
