# 系统方法 API

系统的方法很多，下面只展示几种常用的方法。

# 1. 调用二维码

// 1.1在vue的程序中调用
new window.PowerApp.system.ScanQRCode((result) => {
  // result 是二维码扫描的结果
  this.address = result;
});


// 1.2在vue的程序中iframe嵌入页面的调用
var that = this;
window.parent.PowerApp && 
new window.parent.PowerApp.system.ScanQRCode(function (result) {
  // result 是二维码扫描的结果
  that.address = result;
});

# 2. 上传附件

// 2.1在vue的程序中调用
// 参数：token, KeyValue, KeyWord, successCallBack, errorCallBack
new window.PowerApp.system.UploadFile(
  token, 
  KeyValue, 
  KeyWord, 
  (res) => {
  // successCallBack

  }, (err) => {
  // errorCallBack

});


// 2.2在vue的程序中iframe嵌入页面的调用
var that = this;
window.parent.PowerApp && 
new window.parent.PowerApp.system.UploadFile(
  token, 
  KeyValue, 
  KeyWord, 
  function (res) {
    // successCallBack
  }, function (err) {
    // errorCallBack
  }
);

# 3. alert

// 1在vue的程序中调用
// title, message, callback(选填)
window.PowerApp.system.alert("title", "提示信息", () => {
  // 回调函数
});


// 2在vue的程序中iframe嵌入页面的调用
window.parent.PowerApp && 
window.parent.PowerApp.system.alert("title", "提示信息", function () {
  // 回调函数
});

或者 
// 1在vue的程序中调用
let option = {
  title: "提示",
  message: "你错了，还继续吗？"
};
window.PowerApp.tools.alert(option, () => {
  // 回调函数
});

// 2在vue的程序中iframe嵌入页面的调用
window.parent.PowerApp && 
window.parent.PowerApp.tools.alert(option function () {
  // 回调函数
});

// 2在vue的程序中iframe嵌入页面的调用

# 8. openActionSheet

打开ActionSheet

  { name: "相机", key: 1 },
  { name: "图库", key: 2 }
];
window.PowerApp.tools.openActionSheet(data, (selectItem) => {
  console.log(selectItem);
});


// 2在vue的程序中iframe嵌入页面的调用
window.parent.PowerApp &&
window.parent.PowerApp.tools.openActionSheet(data, function (selectItem) {
  console.log(selectItem);
});

# 13.getSession

获取人员信息 带token

// 1. 在vue的程序中调用
let session = window.PowerApp.tools.getSession();

// 2 在vue的程序中iframe嵌入页面的调用
if (window.parent.PowerApp) {
  var session = window.parent.PowerApp.tools.getSession();
}

// session 的结果
session: {
  token: "",
  UserData: {}
}

# 14.OutGuide 外联向导

// 1. 在vue的程序中调用
window.PowerApp.tools.OutGuide({
  url: "/static/flow/test.html",
  title: "123"
});

// 2 在vue的程序中iframe嵌入页面的调用
window.parent.PowerApp && 
window.parent.PowerApp.tools.OutGuide({
  url: "/static/flow/test.html",
  title: "123"
}, function (res) {
  console.log(res); // 向导数据选中的结果
});
